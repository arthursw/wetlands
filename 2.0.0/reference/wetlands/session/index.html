    class WetlandsSession:
    """Session-owned runtime.

    This bounded implementation owns the current process root marker and adapts
    configuration into the existing manager. Environment creation and worker dispatch are
    implemented by later runtime phases.
    """

    def __init__(self, config: WetlandsConfig | None = None, **options: object) -> None:
        backend = options.pop("_backend", None)
        self.config = WetlandsConfig.from_options(config, **options)
        self._owner: RuntimeOwner | None = RuntimeOwner.acquire(self.config.root, kind="session")
        try:
            self._manager = EnvironmentManager(**self.config.to_environment_manager_kwargs())
        except Exception:
            self._release_owner()
            raise
        self._registry = EnvironmentRegistry(self.config.root)
        self._backend: _EnvironmentBackend = (
            backend
            if backend is not None
            else _ManagerBackend(self._manager, main_env_policy=self.config.main_env_policy)
        )
        self._session_environments: dict[str, _SessionEnvironment] = {}
        self.closed = False

    def __enter__(self) -> WetlandsSession:
        return self

    def __exit__(self, exc_type: object, exc: object, traceback: object) -> None:
        self.close()

    def close(self) -> None:
        if self.closed:
            return
        try:
            self._stop_owned_environments()
        finally:
            self._release_owner()
            self.closed = True

    def env(
        self,
        name: str,
        spec: EnvSpec,
        workers: int = 1,
        on_conflict: ConflictPolicy = "error",
        auto_start: bool = True,
        **options: Any,
    ) -> object:
        validate_environment_name(name)
        self._validate_spec(spec)
        self._validate_workers(workers)
        self._validate_conflict_policy(on_conflict)
        self._validate_auto_start(auto_start)
        self._ensure_root_owned_before_starting(workers)
        _ = options
        entry = self._registry_entry(name)
        path = self._backend.environment_path(name)
        requested_fingerprint = spec.fingerprint()

        if entry is None:
            if self._backend.exists(name):
                path = self._acquire_untracked(name, spec, workers, on_conflict, auto_start)
                ownership: Ownership = "owned" if on_conflict == "recreate" else "accepted-untracked"
            else:
                path, ownership = self._create_owned(name, spec, workers, auto_start)
            return self._environment_handle(name, path, spec, ownership, workers, auto_start)

        path = Path(entry["path"])
        entry_fingerprint = entry.get("spec_fingerprint")
        ownership = entry.get("owner", "owned")
        if ownership not in {"owned", "attached", "accepted-untracked"}:
            ownership = "owned"
        ownership = ownership  # type: ignore[assignment]
        existing_spec = EnvSpec.from_dict(entry["spec"]) if isinstance(entry.get("spec"), dict) else None

        if entry_fingerprint == requested_fingerprint:
            self._validate_existing_entry_for_policy(name, path)
            self._registry.record_acquisition(name, desired_workers=workers, auto_start=auto_start)
            return self._environment_handle(name, path, spec, ownership, workers, auto_start)

        if on_conflict == "error":
            raise EnvironmentConflictError(
                f"Environment '{name}' conflicts with requested spec fingerprint {requested_fingerprint}"
            )
        if on_conflict == "reuse":
            self._validate_existing_entry_for_policy(name, path)
            self._registry.record_acquisition(name, desired_workers=workers, auto_start=auto_start)
            return self._environment_handle(name, path, existing_spec or spec, ownership, workers, auto_start)
        if ownership != "owned":
            raise EnvironmentConflictError(
                f"Environment '{name}' is {ownership}; refusing destructive conflict policy '{on_conflict}'"
            )
        self._validate_existing_entry_for_policy(name, path)
        if on_conflict == "update":
            path = self._update_owned(name, spec, path, workers, auto_start)
            return self._environment_handle(name, path, spec, "owned", workers, auto_start)

        path = self._recreate_owned(name, spec, path, workers, auto_start)
        return self._environment_handle(name, path, spec, "owned", workers, auto_start)

    def attach_env(
        self,
        name: str,
        path: str | Path,
        spec: EnvSpec | None = None,
        workers: int = 0,
        auto_start: bool = True,
        **options: Any,
    ) -> object:
        validate_environment_name(name)
        if spec is not None:
            self._validate_spec(spec)
        self._validate_workers(workers)
        self._validate_auto_start(auto_start)
        self._ensure_root_owned_before_starting(workers)
        _ = options
        environment_path = Path(path)
        if not environment_path.exists():
            raise FileNotFoundError(f"Attached environment path does not exist: {environment_path}")
        self._registry.record_success(
            name,
            path=environment_path,
            spec=spec,
            ownership="attached",
            backend=self._backend.name,
            desired_workers=workers,
            auto_start=auto_start,
        )
        return self._environment_handle(name, environment_path, spec, "attached", workers, auto_start)

    @property
    def manager(self) -> EnvironmentManager:
        return self._manager

    def _release_owner(self) -> None:
        if self._owner is None:
            return
        self._owner.release()
        self._owner = None

    def _ensure_root_owned_before_starting(self, workers: int) -> None:
        if workers == 0:
            return
        if self._owner is None:
            raise RootOwnershipError("Root ownership is required before starting workers")
        marker = self._owner.root / "runtime_owner.json"
        with root_lock(self._owner.root):
            try:
                payload = json.loads(marker.read_text())
            except FileNotFoundError as exc:
                raise RootOwnershipError("Root ownership marker is missing; refusing to start workers") from exc
            except json.JSONDecodeError as exc:
                raise RootOwnershipError("Root ownership marker is invalid; refusing to start workers") from exc
            if payload.get("runtime_id") != self._owner.runtime_id:
                raise RootOwnershipError("Root ownership marker belongs to another runtime; refusing to start workers")

    def _stop_owned_environments(self) -> None:
        for environment in list(self._session_environments.values()):
            environment.stop(force=True)
        environments = getattr(self._manager, "environments", {})
        for name, environment in list(environments.items()):
            if name == "wetlands_main":
                continue
            stop = getattr(environment, "stop", None)
            if callable(stop):
                stop(force=True)

    @staticmethod
    def _validate_spec(spec: EnvSpec) -> None:
        if not isinstance(spec, EnvSpec):
            raise TypeError("spec must be an EnvSpec")

    @staticmethod
    def _validate_workers(workers: int) -> None:
        if not isinstance(workers, int) or isinstance(workers, bool) or workers < 0:
            raise ValueError("workers must be a non-negative integer")

    @staticmethod
    def _validate_conflict_policy(on_conflict: str) -> None:
        if on_conflict not in _CONFLICT_POLICIES:
            raise ValueError("on_conflict must be one of 'error', 'reuse', 'update', or 'recreate'")

    @staticmethod
    def _validate_auto_start(auto_start: bool) -> None:
        if not isinstance(auto_start, bool):
            raise TypeError("auto_start must be a bool")

    def _registry_entry(self, name: str) -> dict[str, object] | None:
        payload = self._registry.load()
        envs = payload["envs"]
        if not isinstance(envs, dict):
            return None
        entry = envs.get(name)
        return dict(entry) if isinstance(entry, dict) else None

    def _remove_registry_entry(self, name: str) -> None:
        validate_environment_name(name)
        with root_lock(self.config.root):
            payload = self._registry.load()
            envs = payload["envs"]
            if isinstance(envs, dict):
                envs.pop(name, None)
            self._registry.save(payload)

    def _record_backend_failure(self, name: str, path: Path, spec: EnvSpec, error: Exception) -> None:
        self._registry.record_failure(name, path=path, spec=spec, error=str(error))

    def _create_owned(self, name: str, spec: EnvSpec, workers: int, auto_start: bool) -> tuple[Path, Ownership]:
        path = self._backend.environment_path(name)
        ownership: Ownership = "owned"
        try:
            path = self._backend.create(name, spec)
            if self._is_main_environment_path(path):
                ownership = "attached"
        except Exception as exc:
            self._record_backend_failure(name, path, spec, exc)
            raise
        self._registry.record_success(
            name,
            path=path,
            spec=spec,
            ownership=ownership,
            backend=self._backend.name,
            desired_workers=workers,
            auto_start=auto_start,
        )
        return path, ownership

    def _update_owned(self, name: str, spec: EnvSpec, path: Path, workers: int, auto_start: bool) -> Path:
        try:
            path = self._backend.update(name, spec)
        except Exception as exc:
            self._record_backend_failure(name, path, spec, exc)
            raise
        self._registry.record_success(
            name,
            path=path,
            spec=spec,
            ownership="owned",
            backend=self._backend.name,
            desired_workers=workers,
            auto_start=auto_start,
        )
        return path

    def _recreate_owned(self, name: str, spec: EnvSpec, path: Path, workers: int, auto_start: bool) -> Path:
        try:
            path = self._backend.recreate(name, spec)
        except Exception as exc:
            self._record_backend_failure(name, path, spec, exc)
            raise
        self._registry.record_success(
            name,
            path=path,
            spec=spec,
            ownership="owned",
            backend=self._backend.name,
            desired_workers=workers,
            auto_start=auto_start,
        )
        return path

    def _acquire_untracked(
        self,
        name: str,
        spec: EnvSpec,
        workers: int,
        on_conflict: ConflictPolicy,
        auto_start: bool,
    ) -> Path:
        path = self._backend.environment_path(name)
        if on_conflict == "error":
            raise EnvironmentConflictError(f"Environment '{name}' exists but is not tracked by Wetlands")
        if on_conflict == "reuse":
            ownership: Ownership = "accepted-untracked"
            self._registry.record_success(
                name,
                path=path,
                spec=spec,
                ownership=ownership,
                backend=self._backend.name,
                desired_workers=workers,
                auto_start=auto_start,
            )
            return path
        if on_conflict == "update":
            try:
                path = self._backend.update(name, spec)
            except Exception as exc:
                self._record_backend_failure(name, path, spec, exc)
                raise
            self._registry.record_success(
                name,
                path=path,
                spec=spec,
                ownership="accepted-untracked",
                backend=self._backend.name,
                desired_workers=workers,
                auto_start=auto_start,
            )
            return path
        return self._recreate_owned(name, spec, path, workers, auto_start)

    def _is_main_environment_path(self, path: Path) -> bool:
        is_main_environment_path = getattr(self._backend, "is_main_environment_path", None)
        return bool(callable(is_main_environment_path) and is_main_environment_path(path))

    def _validate_existing_entry_for_policy(self, name: str, path: Path) -> None:
        if self.config.main_env_policy != "only":
            return
        if self._is_main_environment_path(path):
            return
        raise EnvironmentConflictError(
            f"Environment '{name}' is not backed by the main environment required by main_env_policy='only'"
        )

    def _environment_handle(
        self,
        name: str,
        path: Path,
        spec: EnvSpec | None,
        ownership: Ownership,
        workers: int,
        auto_start: bool,
    ) -> _SessionEnvironment:
        environment = self._session_environments.get(name)
        if environment is None:
            environment = _SessionEnvironment(
                self,
                name=name,
                path=path,
                spec=spec,
                registry_owner=ownership,
                auto_start=auto_start,
            )
            self._session_environments[name] = environment
        else:
            environment.configure(path=path, spec=spec, registry_owner=ownership, auto_start=auto_start)
        previous_auto_start = environment.auto_start
        scale_kwargs = {"auto_start": auto_start} if auto_start is False or auto_start != previous_auto_start else {}
        environment.scale(workers, **scale_kwargs)
        return environment
